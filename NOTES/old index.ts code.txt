// ---------------- Utility: Haversine Distance ----------------
function haversineDistance([lat1, lon1]: number[], [lat2, lon2]: number[]): number {
  const R = 6371000;
  const toRad = (x: number) => (x * Math.PI) / 180;
  const dLat = toRad(lat2 - lat1);
  const dLon = toRad(lon2 - lon1);
  const a =
    Math.sin(dLat / 2) ** 2 +
    Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.sin(dLon / 2) ** 2;
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  return R * c;
}

// ---------------- Storage Helper ----------------
async function uploadBase64FileToBucket(base64Data: string, path: string, bucket: any): Promise<string> {
  const buffer = Buffer.from(base64Data, "base64");
  const file = bucket.file(path);
  await file.save(buffer, { resumable: false });
  await file.makePublic();
  return file.publicUrl();
}

    "pest": "Public Health / Entomology",
    "fever": "Public Health / Entomology",
    "fogging": "Public Health / Entomology",
  };

  const matchedDepartments = new Set<string>();
  const matchedKeywords: string[] = [];

    return { department: [...matchedDepartments][0], tags: matchedKeywords };
  }

  // ❌ 2. No matches OR multiple matches → trigger Gemini fallback
  const nlpPrediction = await callGeminiNLP(transcript);
  return nlpPrediction;
}

// ---------------- Audio STT + Translation ----------------
async function transcribeAudioInternal(audioUrl: string): Promise<string> {
  if (!audioUrl) throw new functions.https.HttpsError("failed-precondition", "No audio to transcribe");

  try {
    const [response] = await speechClient.recognize({
      audio: { uri: audioUrl },
      config: {
        encoding: "MP3",
        languageCode: "te-IN", // Telugu default
        alternativeLanguageCodes: ["en-US", "hi-IN"], // English + Hindi fallback
      },
    });

    let transcript = response.results?.map(r => r.alternatives?.[0].transcript).join(" ") || "";
    const detectedLanguage = response.results?.[0]?.languageCode || "en-US";

    let translatedTranscript = transcript;

    // 🌐 If detected language ≠ English → translate
    if (detectedLanguage !== "en-US" && transcript) {
      const [translation] = await translateClient.translateText({
        contents: [transcript],
        targetLanguageCode: "en",
        parent: `projects/${process.env.GOOGLE_PROJECT_ID}/locations/global`,
      });

      translatedTranscript = translation.translations?.[0]?.translatedText || transcript;
    }

    return translatedTranscript;
  } catch (error) {
    functions.logger.error("STT Translation Error:", error);
    throw new functions.https.HttpsError("internal", "Audio transcription failed");
  }
}

// ---------------- Create Complaint ----------------
export const createComplaint = functions.https.onCall(async (data: any) => {
  const { userId, transcript, audioBase64, photoBase64, gpsLat, gpsLong, tags } = data as CreateComplaintData;

  // 🔒 Validation
  if (!userId || !photoBase64 || gpsLat === undefined || gpsLong === undefined) {
    throw new functions.https.HttpsError("invalid-argument", "Photo and location are mandatory");
  }

  if (!transcript && !audioBase64) {
    throw new functions.https.HttpsError("invalid-argument", "Either transcript or audio must be provided");
  }

  // 🖼️ Upload Image
  const photoPath = `complaints/${Date.now()}_${userId}.jpg`;
  const photoUrl = await uploadBase64FileToBucket(photoBase64, photoPath, complaintImagesBucket);

  let finalTranscript = transcript || "";
  let audioUrl = "";
  let department = "";
  let finalTags: string[] = tags || [];

  // 🎙️ STT runs only when transcript missing
  if (!transcript && audioBase64) {
    const audioPath = `complaints/audio_${Date.now()}_${userId}.mp3`;
    audioUrl = await uploadBase64FileToBucket(audioBase64, audioPath, complaintAudioBucket);
    finalTranscript = await transcribeAudioInternal(audioUrl);
  }

  // 🧠 Department prediction
  const prediction = await predictDepartment(finalTranscript);
  department = prediction.department;
  finalTags = prediction.tags;

  // 🔁 Duplicate check (same dept + Open + ≤ 250m)
  const complaintsRef = db.collection("complaints");
  const snapshot = await complaintsRef
    .where("status", "==", "Open")
    .where("department", "==", department)
    .get();

  for (const doc of snapshot.docs) {
    const c = doc.data();
    const distance = haversineDistance([gpsLat, gpsLong], [c.gps_lat, c.gps_long]);
    if (distance <= 250) {
      await doc.ref.update({ report_count: admin.firestore.FieldValue.increment(1) });
      return { complaintId: doc.id, merged: true };
    }
  }

  // 🆕 Add New Complaint
  const newComplaint = {
    user_id: userId,
    transcript: finalTranscript,
    transcript_translated: finalTranscript,
    department,
    tags: finalTags,
    gps_lat: gpsLat,
    gps_long: gpsLong,
    photo_url: photoUrl,
    audio_url: transcript ? "" : audioUrl,
    status: "Open",
    report_count: 1,
    created_at: admin.firestore.Timestamp.now(),
    updated_at: admin.firestore.Timestamp.now(),
  };

  const docRef = await complaintsRef.add(newComplaint);
  return { complaintId: docRef.id, merged: false };
});

// ---------------- Resolve Complaint ----------------
export const resolveComplaint = functions.https.onCall(async (data: any) => {
  const { complaintId, authorityId, resolutionImagesBase64 } = data as ResolveComplaintData;

  if (!complaintId || !authorityId || !resolutionImagesBase64?.length) {
    throw new functions.https.HttpsError("invalid-argument", "Missing required fields");
  }

  const complaintRef = db.collection("complaints").doc(complaintId);
  const complaintSnap = await complaintRef.get();
  if (!complaintSnap.exists) throw new functions.https.HttpsError("not-found", "Complaint not found");

  const imageUrls: string[] = [];
  for (let i = 0; i < resolutionImagesBase64.length; i++) {
    const path = `resolutions/${complaintId}/img_${i}_${Date.now()}.jpg`;
    const url = await uploadBase64FileToBucket(resolutionImagesBase64[i], path, resolutionImagesBucket);
    imageUrls.push(url);
  }

  await complaintRef.update({
    status: "Resolved",
    updated_at: admin.firestore.Timestamp.now(),
  });

  await db.collection("resolutions").add({
    complaint_id: complaintId,
    authority_id: authorityId,
    images: imageUrls,
    created_at: admin.firestore.Timestamp.now(),
  });

  return { success: true, complaintId, resolutionImages: imageUrls };
});
